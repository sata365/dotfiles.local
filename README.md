export GIT_BASE_DIR=/path/to/git/repository
export LOCAL_DOTFILES_REPO=https://your.local.dotfiles/repository
# Install
/bin/bash -c "$(curl -fsSL https://go.saveourservers.io/dotfiles-macos)"
